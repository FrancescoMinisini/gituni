cmake_minimum_required(VERSION 3.15)

# Nome del progetto
project(MainROOTProgram)

# Imposta la versione standard di C++
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Trova il pacchetto ROOT
find_package(ROOT REQUIRED)

# Aggiungi l'eseguibile
add_executable(main main.cpp)

# Collega le librerie ROOT all'eseguibile
target_include_directories(main PUBLIC ${ROOT_INCLUDE_DIRS})
target_link_libraries(main PUBLIC ${ROOT_LIBRARIES})

# Abilita i flag consigliati di ROOT
target_compile_options(main PUBLIC ${ROOT_CXX_FLAGS})
target_link_options(main PUBLIC ${ROOT_LD_FLAGS})

# Messaggi informativi
message(STATUS "ROOT include dirs: ${ROOT_INCLUDE_DIRS}")
message(STATUS "ROOT libraries: ${ROOT_LIBRARIES}")
